Exception Handling

Input Validation with try/catch/throw/finally

Enter a number between 1 and 100:

Code Sample

function validateNumber(input) {
    try {
        if (input === "") throw "Input is empty!";
        var num = parseFloat(input);
        if (isNaN(num)) throw "Not a valid number!";
        if (num < 1) throw "Number too low! Must be >= 1";
        if (num > 100) throw "Number too high! Must be <= 100";
        return "Valid input: " + num;
    } catch (error) {
        return "Error: " + error;
    } finally {
        console.log("Validation attempt complete.");
    }
}
