How It Works

<!-- Embed XML as a data island -->
<xml id="studentData">
    <students>
        <student id="S001">
            <name>Alice</name>
            <age>21</age>
        </student>
    </students>
</xml>

<script>
var xmlDoc = document.getElementById("studentData").XMLDocument;
var students = xmlDoc.getElementsByTagName("student");
for (var i = 0; i < students.length; i++) {
    var name = students[i].getElementsByTagName("name")[0].text;

}
</script>
